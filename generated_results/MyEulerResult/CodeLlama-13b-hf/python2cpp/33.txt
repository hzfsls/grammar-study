int digitFactorials(int n) {
    int result = 0;
    for (int i = 3; i < n; i++) {
        int fact_sum = 0;
        for (char digit : to_string(i)) {
            int fact = 1;
            for (int j = 1; j <= digit - '0'; j++) {
                fact *= j;
            }
            fact_sum += fact;
        }
        if (i == fact_sum) {
            result += i;
        }
    }
    return result;
}
