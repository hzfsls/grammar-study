function codedTriangleNumbers($words) {
    $result = 0;
    foreach ($words as $word) {
        $value = 0;
        for ($i = 0; $i < strlen($word); $i++) {
            $value += ord($word[$i]) - 64;
        }
        $n = intval(sqrt($value * 2));
        if ($n * ($n + 1) == $value * 2) {
            $result++;
        }
    }
    return $result;
}
