def pandigitalMultiples(n: Int): Int = {
  var result = -1
  for (i <- 2 until n + 1) {
    var cprod = ""
    for (j <- 1 until 10) {
      cprod += i * j
      if (cprod.length == 9) {
        if (cprod.sorted.mkString == "123456789") {
          result = math.max(result, cprod.toInt)
          break
        }
      } else if (cprod.length > 9) {
        break
      }
    }
  }
  result
}
