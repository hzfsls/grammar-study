const longestCollatzSequence = (n) => {
    let longest = 0;
    let result = 0;
    for (let i = 1; i < n; i++) {
        let chain = 1;
        let num = i;
        while (num!== 1) {
            if (num % 2 === 0) {
                num = num / 2;
            } else {
                num = 3 * num + 1;
            }
            chain += 1;
        }
        if (chain > longest) {
            longest = chain;
            result = i;
        }
    }
    return result;
}
