const namesScores = (names, queries) => {
    let sNames = slices.Clone(names);
    slices.Sort(sNames);
    let result = 0;
    for (let i = 0; i < sNames.length; i++) {
        let x = 0;
        for (let c of sNames[i]) {
            x += c.charCodeAt(0) - 64;
        }
        for (let query of queries) {
            if (query === sNames[i]) {
                result += x * (i + 1);
                break;
            }
        }
    }
    return result;
}
