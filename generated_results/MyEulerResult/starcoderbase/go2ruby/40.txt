def is_prime(n)
    if n < 2
        return false
    end
    if n == 2
        return true
    end
    if n % 2 == 0
        return false
    end
    for i in 3..Math.sqrt(n).to_i
        if n % i == 0
            return false
        end
    end
    return true
end

def pandigital_prime(n)
    for i in n - 1..1
        if is_prime(i)
            si = i.to_s
            length = si.length
            flag = true
            for j in 1..length
                if!si.include?(j.to_s)
                    flag = false
                    break
                end
            end
            if flag
                return i
            end
        end
    end
    return -1
end
