def is_palindrome(s)
    for i in 0...(s.length / 2)
        return false if s[i]!= s[s.length - i - 1]
    end
    true
end

def largest_palindrome_product(n)
    result = 0
    (100...1000).each do |i|
        (i...1000).each do |j|
            prod = i * j
            if is_palindrome(prod.to_s) && prod > result && prod < n
                result = prod
            end
        end
    end
    result
end
