fun smallestMultiple(n: Int): Int {
    var result = 1
    for (i in 1..n) {
        if (result % i!= 0) {
            for (j in 1..n) {
                if ((result * j) % i == 0) {
                    result *= j
                    break
                }
            }
        }
    }
    return result
}
