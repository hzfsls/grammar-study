fun codedTriangleNumbers(words: List<String>): Int {
    var result = 0
    for (word in words) {
        var value = 0
        for (c in word) {
            value += c.toInt() - 64
        }
        val n = kotlin.math.sqrt(value * 2.0).toInt()
        if (n * (n + 1) == value * 2) {
            result++
        }
    }
    return result
}
