func IsPalindrome(s string) bool {
    for i := 0; i < len(s) / 2; i++ {
        if s[i]!= s[len(s)-1-i] {
            return false
        }
    }
    return true
}

func DoubleBasePalindromes(n int) int {
    result := 0
    for i := 1; i < n; i++ {
        strI := strconv.Itoa(i)
        binI := strconv.FormatInt(int64(i), 2)
        if IsPalindrome(strI) && IsPalindrome(binI) {
            result += i
        }
    }
    return result
}
