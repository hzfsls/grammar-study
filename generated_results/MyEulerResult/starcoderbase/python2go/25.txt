func ReciprocalCycles(n int) int {
    result := 0
    max_length := 0
    for i := 1; i < n; i++ {
        remainders := []int{}
        remainder := 1
        for remainder!= 0 && remainder!= 1 {
            remainders = append(remainders, remainder)
            remainder = (remainder * 10) % i
        }
        length := 0
        if remainder!= 0 {
            length = len(remainders) - remainders[len(remainders)-1]
        }
        if length > max_length {
            max_length = length
            result = i
        }
    }
    return result
}
